function countOccurrences(arr, n, x):
    # Step 1: Find the first occurrence of x
    first = findFirstOccurrence(arr, n, x)
    
    # If x is not present in the array
    if first == -1:
        return 0  
    
    # Step 2: Find the last occurrence of x
    last = findLastOccurrence(arr, n, x)
    
    # Step 3: Calculate the number of occurrences
    return (last - first + 1)

function findFirstOccurrence(arr, n, x):
    low = 0
    high = n - 1
    result = -1
    
    while low <= high:
        mid = (low + high) / 2
        
        if arr[mid] == x:
            result = mid      # Record the potential first occurrence
            high = mid - 1    # Search left to find the actual first occurrence
        elif arr[mid] > x:
            high = mid - 1
        else:
            low = mid + 1
    
    return result  # Return the index of the first occurrence or -1 if not found

function findLastOccurrence(arr, n, x):
    low = 0
    high = n - 1
    result = -1
    
    while low <= high:
        mid = (low + high) / 2
        
        if arr[mid] == x:
            result = mid      # Record the potential last occurrence
            low = mid + 1     # Search right to find the actual last occurrence
        elif arr[mid] > x:
            high = mid - 1
        else:
            low = mid + 1
    
    return result  # Return the index of the last occurrence or -1 if not found
